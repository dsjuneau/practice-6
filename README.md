# Linux commands

## Find your current directory and what is inside it

pwd
ls
ls -a
ls -l

## Change directory and dot commands

cd bob/ "moves to the bob directory"
cd .. "goes up one"
../../ goes up two

## Add a file

touch app.js

## Remove a file

rm app.js

## Add a directory

mkdir name

## Remove a directory

rm -r directory

## Autocomplete a word

"tab"

## Pull up last command

"arrow up"

## Escape Characters

joe\ and\ bob
